cache_peliculas = {}
 
def obtener_titulo_pelicula(url_film):

    if url_film not in cache_peliculas:
        response = requests.get(url_film)
        if response.status_code == 200:
            cache_peliculas[url_film] = response.json().get('title')
        else:
            return "Desconocida"
    return cache_peliculas[url_film]
